let isAllInOneRunning = false; async function captureAllInOne() { const btn = event.currentTarget; // If already running → Stop it if (isAllInOneRunning) { isAllInOneRunning = false; log('All-in-One stopped by user', 'warn'); showStatus('Scanning stopped', 'text-amber-400'); resetButtonState(btn); hideScanGuide(); return; } // Confirm before clearing existing text if (!confirmClearIfNeeded()) { return; } valueInput.value = ''; // Clear old value isAllInOneRunning = true; setButtonScanning(btn, '⏳ Scanning... (Click again to Stop)'); showScanGuide('all'); const frameCanvas = getCurrentFrameCanvas(); if (!frameCanvas) { isAllInOneRunning = false; resetButtonState(btn); hideScanGuide(); return; } log('All-in-One started (max 100 attempts, ~180ms interval)', 'info'); for (let attempt = 1; attempt <= 100; attempt++) { if (!isAllInOneRunning) break; try { // Try QR const ctx = frameCanvas.getContext('2d', { willReadFrequently: true }); const imageData = ctx.getImageData(0, 0, frameCanvas.width, frameCanvas.height); const qrResult = jsQR(imageData.data, imageData.width, imageData.height); if (qrResult && qrResult.data && qrResult.data.length > 0) { valueInput.value = qrResult.data; log(`All-in-One: QR detected (attempt ${attempt})`, 'success'); showStatus('✅ QR captured', 'text-emerald-400'); clearStatusAfter(2000); isAllInOneRunning = false; resetButtonState(btn); hideScanGuide(); return; } // Try Barcode / Data Matrix const preprocessed = preprocessCanvas(frameCanvas); if (!codeReader) codeReader = new ZXing.BrowserMultiFormatReader(); const result = await tryDecodeWithRotations(preprocessed, codeReader); if (result && result.text && result.text.length > 0) { valueInput.value = result.text; log(`All-in-One: Code detected (attempt ${attempt})`, 'success'); showStatus('✅ Code captured', 'text-emerald-400'); clearStatusAfter(2000); isAllInOneRunning = false; resetButtonState(btn); hideScanGuide(); return; } } catch (e) {} // Increased delay for better reliability await new Promise(resolve => setTimeout(resolve, 380)); } // All attempts exhausted if (isAllInOneRunning) { log('All-in-One: No code found after 100 attempts', 'warn'); showStatus('No code found after 100 attempts', 'text-amber-400'); } isAllInOneRunning = false; resetButtonState(btn); hideScanGuide(); }
